32  /**
33   * The server part of an AF_UNIX domain socket.
34   * 
35   * @author Christian Kohlschütter

37  public class AFUNIXServerSocket extends ServerSocket implements FileDescriptorAccess {

54    AFUNIXServerSocket(FileDescriptor fdObj) throws IOException {
55      super();

69      return new AFUNIXServerSocket(null);

72    public static AFUNIXServerSocket newInstance(FileDescriptor fdObj, int localPort, int remotePort)
73        throws IOException {
74      if (fdObj == null) {
75        return newInstance();
76      }
77  
78      int status = NativeUnixSocket.socketStatus(fdObj);
79      if (!fdObj.valid() || status == NativeUnixSocket.SOCKETSTATUS_INVALID) {
80        throw new SocketException("Not a valid socket");
81      }
82      AFUNIXServerSocket socket = new AFUNIXServerSocket(fdObj);
83      socket.getAFImpl().updatePorts(localPort, remotePort);
84  
85      switch (status) {
86        case NativeUnixSocket.SOCKETSTATUS_CONNECTED:
87          throw new SocketException("Not a ServerSocket");
88        case NativeUnixSocket.SOCKETSTATUS_BOUND:
89          socket.bind(AFUNIXSocketAddress.INTERNAL_DUMMY_BIND);
90  
91          socket.setBoundEndpoint(AFUNIXSocketAddress.getSocketAddress(fdObj, false, localPort));
92          break;
93        case NativeUnixSocket.SOCKETSTATUS_UNKNOWN:
94          break;
95        default:
96          throw new IllegalStateException("Invalid socketStatus response: " + status);
97      }
98  
99      socket.getAFImpl().setSocketAddress(socket.getLocalSocketAddress());
100     return socket;
101   }

179       }

183   @Override
184   public void bind(SocketAddress endpoint, int backlog) throws IOException {
185     if (isClosed()) {
186       throw new SocketException("Socket is closed");
187     }
188     if (isBound()) {
189       throw new SocketException("Already bound");
190     }
191 
192     if (!(endpoint instanceof AFUNIXSocketAddress)) {
193       throw new IllegalArgumentException("Can only bind to endpoints of type "
194           + AFUNIXSocketAddress.class.getName());
195     }
196 
197     getAFImpl().bind(endpoint, getReuseAddress() ? -1 : 0);
198     setBoundEndpoint((AFUNIXSocketAddress) endpoint);
199 
200     if (endpoint == AFUNIXSocketAddress.INTERNAL_DUMMY_BIND) { // NOPMD
201       return;
202     }
203 
204     implementation.listen(backlog);
205   }

207   @Override
208   public boolean isBound() {
209     return boundEndpoint != null;
210   }

212   @Override
213   public boolean isClosed() {
214     return super.isClosed() || (isBound() && !implementation.getFD().valid());
215   }

217   @Override
218   public AFUNIXSocket accept() throws IOException {
219     AFUNIXSocket as = newSocketInstance();
220     boolean success = implementation.accept0(as.getAFImpl());
221     if (isClosed()) {
222       // We may have connected to the socket to unblock it
223       throw new SocketException("Socket is closed");
224     }
225 
226     if (!success) {
227       // non-blocking socket, nothing to accept
228       return null;
229     }
230     as.connect(AFUNIXSocketAddress.INTERNAL_DUMMY_CONNECT);
231     as.getAFImpl().updatePorts(getAFImpl().getLocalPort1(), getAFImpl().getRemotePort());
232 
233     return as;
234   }

242     return "AFUNIXServerSocket[" + (isBound() ? boundEndpoint.toString() : "unbound") + "]";

245   @Override
246   public synchronized void close() throws IOException {
247     if (isClosed()) {
248       return;
249     }
250 
251     AFUNIXSocketAddress endpoint = boundEndpoint;
252 
253     IOException superException = null;
254     try {
255       super.close();
256     } catch (IOException e) {
257       superException = e;
258     }
259     if (implementation != null) {
260       try {
261         implementation.close();
262       } catch (IOException e) {
263         if (superException == null) {
264           superException = e;
265         } else {
266           superException.addSuppressed(e);
267         }
268       }
269     }
270 
271     IOException ex = null;
272     try {
273       closeables.close(superException);
274     } finally {
275       if (endpoint != null && endpoint.hasFilename() && isDeleteOnClose()) {
276         File f = endpoint.getFile();
277         if (!f.delete() && f.exists()) {
278           ex = new IOException("Could not delete socket file after close: " + f);
279         }
280       }
281     }
282     if (ex != null) {
283       throw ex;
284     }
285   }

326     if (boundEndpoint == null) {

333   /**
334    * Checks if this {@link AFUNIXServerSocket}'s file should be removed upon {@link #close()}.
335    * 
336    * Deletion is not guaranteed, especially when not supported (e.g., addresses in the abstract
337    * namespace).
338    * 
339    * @return {@code true} if an attempt is made to delete the socket file upon {@link #close()}.
340    */
341   public boolean isDeleteOnClose() {
342     return deleteOnClose.get();
343   }
344 
345   /**
346    * Enables/disables deleting this {@link AFUNIXServerSocket}'s file upon {@link #close()}.
347    * 
348    * Deletion is not guaranteed, especially when not supported (e.g., addresses in the abstract
349    * namespace).
350    * 
351    * @param b Enabled if {@code true}.
352    */
353   public void setDeleteOnClose(boolean b) {
354     deleteOnClose.set(b);
355   }

357   AFUNIXSocketImpl getAFImpl() {
358     if (created.compareAndSet(false, true)) {
359       try {
360         getSoTimeout(); // trigger create via java.net.Socket
361       } catch (IOException e) {
362         // ignore
363       }
364     }
365     return implementation;
366   }
