128   @Override
129   public void connect(SocketAddress endpoint, int timeout) throws IOException {
130     if (endpoint == null) {
131       throw new IllegalArgumentException("connect: The address can't be null");
132     }
133     if (timeout < 0) {
134       throw new IllegalArgumentException("connect: timeout can't be negative");
135     }
136     if (isClosed()) {
137       throw new SocketException("Socket is closed");
138     }
139 
140     if (!(endpoint instanceof AFUNIXSocketAddress)) {
141       if (socketFactory != null) {
142         if (endpoint instanceof InetSocketAddress) {
143           InetSocketAddress isa = (InetSocketAddress) endpoint;
144 
145           String hostname = isa.getHostString();
146           if (socketFactory.isHostnameSupported(hostname)) {
147             endpoint = socketFactory.addressFromHost(hostname, isa.getPort());
148           }
149         }
150       }
151       if (!(endpoint instanceof AFUNIXSocketAddress)) {
152         throw new IllegalArgumentException("Can only connect to endpoints of type "
153             + AFUNIXSocketAddress.class.getName() + ", got: " + endpoint);
154       }
155     }
156     impl.connect(endpoint, timeout);
157     this.addr = (AFUNIXSocketAddress) endpoint;
158     NativeUnixSocket.setBound(this);
159     NativeUnixSocket.setConnected(this);
160   }

162   @Override
163   public String toString() {
164     if (isConnected()) {
165       return "AFUNIXSocket[fd=" + impl.getFD() + ";addr=" + addr.toString() + "]";
166     }
167     return "AFUNIXSocket[unconnected]";
168   }

293   private static synchronized int getCapabilities() {
294     if (capabilities == null) {
295       if (!isSupported()) {
296         capabilities = 0;
297       } else {
298         capabilities = NativeUnixSocket.capabilities();
299       }
300     }
301     return capabilities.intValue();
302   }
303 
304   /**
305    * Checks if the current environment (system platform, native library, etc.) supports a given
306    * junixsocket capability.
307    * 
308    * @param capability The capability.
309    * @return true if supported.
310    */
311   public static boolean supports(AFUNIXSocketCapability capability) {
312     return (getCapabilities() & capability.getBitmask()) != 0;
313   }

315   @Override
316   public synchronized void close() throws IOException {
317     IOException superException = null;
318     try {
319       super.close();

321       superException = e;
322     }
323     closeables.close(superException);
324   }
