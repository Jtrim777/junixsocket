18  package org.newsclub.net.unix;
19  
20  import java.io.FileDescriptor;
21  import java.io.IOException;
22  import java.nio.ByteBuffer;
23  import java.util.concurrent.atomic.AtomicLong;
24  
25  /**
26   * A shared core that is common for all AFUNIX sockets (datagrams, streams).
27   * 
28   * @author Christian Kohlschütter
29   */
30  class AFUNIXSocketCore extends AFUNIXCore {
31    private static final int SHUT_RD_WR = 2;
32  
33    /**
34     * We keep track of the server's inode to detect when another server connects to our address.
35     */
36    protected final AtomicLong inode = new AtomicLong(-1);
37  
38    protected AFUNIXSocketAddress socketAddress;
39  
40    protected AFUNIXSocketCore(Object observed, FileDescriptor fd,
41        AncillaryDataSupport ancillaryDataSupport) {
42      super(observed, fd, ancillaryDataSupport);
43    }
44  
45    @Override
46    protected void doClose() throws IOException {
47      NativeUnixSocket.shutdown(fd, SHUT_RD_WR);
48      unblockAccepts();
49  
50      super.doClose();
51    }
52  
53    protected void unblockAccepts() {
54    }
55  
56    AFUNIXSocketAddress receive(ByteBuffer dst) throws IOException {
57      ByteBuffer socketAddressBuffer = AFUNIXSocketAddress.SOCKETADDRESS_BUFFER_TL.get();
58      int read = read(dst, socketAddressBuffer, 0);
59      if (read > 0) {
60        return AFUNIXSocketAddress.ofInternal(socketAddressBuffer);
61      } else {
62        return null;
63      }
64    }
65  
66    boolean isConnected(boolean boundOk) {
67      try {
68        if (fd.valid()) {
69          switch (NativeUnixSocket.socketStatus(fd)) {
70            case NativeUnixSocket.SOCKETSTATUS_CONNECTED:
71              return true;
72            case NativeUnixSocket.SOCKETSTATUS_BOUND:
73              if (boundOk) {
74                return true;
75              }
76              break;
77            default:
78          }
79        }
80      } catch (IOException e) {
81        throw new IllegalStateException(e);
82      }
83      return false;
84    }
85  }
