18  package org.newsclub.net.unix.demo.rmi.fd;
19  
20  import org.newsclub.net.unix.rmi.AFUNIXNaming;
21  
22  /**
23   * Demonstrates how to read/write files via FileDescriptors that are exchanged via RMI.
24   * 
25   * This allows reading/writing from and to files that are otherwise not even accessible by the user.
26   * For example, starting the {@link StreamServer} as root and the {@link StreamClient} as a
27   * non-privileged user will allow the non-privileged user to read files only accessible to root.
28   * 
29   * NOTE: For obvious security reasons, running this server without modification is not advised for
30   * anything other than demo purposes.
31   * 
32   * @author Christian Kohlschütter
33   * @see StreamClient
34   */
35  public class StreamServer {
36    public static void main(String[] args) throws Exception {
37      AFUNIXNaming naming = AFUNIXNaming.getInstance();
38      System.out.println("Socket directory: " + naming.getSocketFactory().getSocketDir());
39  
40      try (StreamServiceImpl service = new StreamServiceImpl(naming.getSocketFactory())) {
41        naming.exportAndBind("streamService", service);
42  
43        System.out.println("StreamServer ready; user.name=" + System.getProperty("user.name"));
44      }
45    }
46  }
