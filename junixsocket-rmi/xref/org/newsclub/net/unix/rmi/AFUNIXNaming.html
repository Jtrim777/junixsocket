73    private AFUNIXNaming(final File socketDir, final int port, final String socketPrefix,
74        final String socketSuffix) throws IOException {
75      super();
76      this.registrySocketDir = socketDir;
77      this.registryPort = port;
78      this.servicePort = AFUNIXRMIPorts.RMI_SERVICE_PORT;
79      this.socketFactory = new AFUNIXRMISocketFactory(this, socketDir, null, null, socketPrefix,

83    /**
84     * Returns a new private instance that resides in a custom location, to avoid any collisions with
85     * existing instances.
86     * 
87     * @return The private {@link AFUNIXNaming} instance.
88     * @throws IOException if the operation fails.
89     */
90    public static AFUNIXNaming newPrivateInstance() throws IOException {
91      File tmpDir = Files.createTempDirectory("junixsocket-").toFile();
92      if (!tmpDir.canWrite()) {
93        throw new IOException("Could not create temporary directory: " + tmpDir);
94      }
95      AFUNIXNaming instance = getInstance(tmpDir, AFUNIXRMIPorts.DEFAULT_REGISTRY_PORT);
96      synchronized (AFUNIXNaming.class) {
97        instance.deleteRegistrySocketDir = true;
98      }
99      return instance;
100   }

142   public static AFUNIXNaming getInstance(File socketDir, final int registryPort,
143       String socketPrefix, String socketSuffix) throws RemoteException {
144     Objects.requireNonNull(socketDir);
145     final AFUNIXNamingRef sap = new AFUNIXNamingRef(socketDir, registryPort, socketPrefix,
146         socketSuffix);
147     AFUNIXNaming instance;
148     synchronized (AFUNIXNaming.class) {
149       instance = INSTANCES.get(sap);
150       if (instance == null) {
151         try {
152           instance = new AFUNIXNaming(sap.socketDir, registryPort, socketPrefix, socketSuffix);
153         } catch (RemoteException e) {
154           throw e;
155         } catch (IOException e) {
156           throw new RemoteException(e.getMessage(), e);
157         }
158         INSTANCES.put(sap, instance);
159       }
160     }
161     return instance;
162   }

220   AFUNIXRMIService getRMIService(AFUNIXRegistry reg) throws RemoteException, NotBoundException {
221     if (rmiService == null) {
222       this.rmiService = getRMIServiceFromRegistry(reg);
223     }
224     return rmiService;
225   }

227   AFUNIXRMIService getRMIServiceFromRegistry(AFUNIXRegistry reg) throws RemoteException,
228       NotBoundException {
229     AFUNIXRMIService service;
230     service = (AFUNIXRMIService) reg.lookup(RMI_SERVICE_NAME, 5, TimeUnit.SECONDS);
231     this.remoteShutdownAllowed = service.isShutdownAllowed();
232     return service;
233   }

235   private void closeUponRuntimeShutdown() {
236     if (addedShutdownHook.compareAndSet(false, true)) {
237       ShutdownHookSupport.addWeakShutdownHook(new ShutdownHook() {
238 
239         @Override
240         public void onRuntimeShutdown(Thread thread) throws IOException {
241           synchronized (AFUNIXNaming.class) {
242             if (registry != null && registry.isLocal()) {
243               shutdownRegistry();
244             }
245           }
246         }
247       });
248     }
249   }

252     rmiService = assigner;
253     getRegistry().rebind(RMI_SERVICE_NAME, assigner);

261   /**
262    * Returns a reference to the existing RMI registry.
263    * 
264    * If there's no registry running at this port after waiting for up to the given time, an
265    * exception is thrown.
266    * 
267    * @param timeout The timeout value.
268    * @param unit The timeout unit.
269    * @return The registry.
270    * @throws RemoteException If there was a problem.
271    */
272   public AFUNIXRegistry getRegistry(long timeout, TimeUnit unit) throws RemoteException {
273     if (shutdownInProgress.get()) {
274       throw new ShutdownException();
275     }
276     synchronized (AFUNIXNaming.class) {
277       AFUNIXRegistry reg = getRegistry(false);
278       if (reg == null) {
279         File socketFile = getRegistrySocketFile();
280         if (!socketFile.exists()) {
281           if (waitUntilFileExists(socketFile, timeout, unit)) {
282             reg = getRegistry(false);
283             if (reg != null) {
284               return reg;
285             }
286           }
287         }
288         throw new ShutdownException("Could not find registry at " + getRegistrySocketFile());
289       }
290       return reg;
291     }
292   }
293 
294   private boolean waitUntilFileExists(File f, long timeout, TimeUnit unit) {
295     long timeWait = unit.toMillis(timeout);
296 
297     try {
298       while (timeWait > 0 && !f.exists()) {
299         Thread.sleep(Math.min(50, timeWait));
300         timeWait -= 50;
301       }
302     } catch (InterruptedException e) {
303       // ignored
304     }
305 
306     return f.exists();
307   }

309   /**
310    * Returns a reference to the RMI registry, or {@code null}.
311    *
312    * If there's no registry running at this port, and {@code create} is set to {@code true}, a new
313    * one is created; when {@code create} is set to {@code false}, {@code null} is returned.
314    * 
315    * @param create {@code true} if a new register may be created if necessary.
316    * @return The registry, or {@code null}
317    * @throws RemoteException If there was a problem.
318    */
319   public AFUNIXRegistry getRegistry(boolean create) throws RemoteException {
320     if (shutdownInProgress.get()) {
321       throw new ShutdownException();
322     }
323     synchronized (AFUNIXNaming.class) {
324       if (registry != null) {
325         return registry;
326       } else if (!socketFactory.hasSocketFile(registryPort)) {
327         return create ? createRegistry() : null;
328       }
329 
330       AFUNIXRegistry reg = locateRegistry();
331       setRegistry(reg);
332 
333       try {
334         getRMIService(reg);
335       } catch (NotBoundException | NoSuchObjectException | ConnectIOException e) {
336         if (create) {
337           setRegistry(null);
338           return createRegistry();
339         } else {
340           throw new ServerException("Could not access " + AFUNIXRMIService.class.getName(), e);
341         }
342       }
343 
344       return registry;
345     }
346   }

348   private AFUNIXRegistry locateRegistry() throws RemoteException {
349     Registry regImpl = LocateRegistry.getRegistry(null, registryPort, socketFactory);
350     return regImpl == null ? null : new AFUNIXRegistry(this, regImpl);
351   }

353   /**
354    * Shuts this RMI Registry down.
355    * 
356    * @throws RemoteException if the operation fails.
357    */
358   public void shutdownRegistry() throws RemoteException {
359     synchronized (AFUNIXNaming.class) {
360       if (registry == null) {
361         return;
362       }
363 
364       AFUNIXRegistry registryToBeClosed = registry;
365       AFUNIXRMIService rmiServiceToBeClosed = rmiService;
366 
367       if (!registryToBeClosed.isLocal()) {
368         if (!isRemoteShutdownAllowed()) {
369           throw new ServerException("The server refuses to be shutdown remotely");
370         }
371         setRegistry(null);
372 
373         try {
374           shutdownViaRMIService(registryToBeClosed, rmiServiceToBeClosed);
375         } catch (Exception e) {
376           // ignore
377         }
378         return;
379       }
380 
381       setRegistry(null);
382 
383       if (!shutdownInProgress.compareAndSet(false, true)) {
384         return;
385       }
386       try {
387         unexportRMIService(registryToBeClosed, (AFUNIXRMIServiceImpl) rmiServiceToBeClosed);
388         forceUnexportBound(registryToBeClosed);
389         closeSocketFactory();
390         deleteSocketDir();
391 
392       } finally {
393         shutdownInProgress.set(false);
394       }
395     }
396   }

398   private void unexportRMIService(AFUNIXRegistry reg, AFUNIXRMIServiceImpl serv)
399       throws AccessException, RemoteException {
400     if (serv != null) {
401       serv.shutdownRegisteredCloseables();
402     }
403 
404     try {
405       if (serv != null) {
406         unexportObject(serv);
407       }
408       reg.unbind(RMI_SERVICE_NAME);
409     } catch (ShutdownException | NotBoundException e) {
410       // ignore
411     }
412     this.rmiService = null;
413   }
414 
415   private void forceUnexportBound(AFUNIXRegistry reg) {
416     try {
417       reg.forceUnexportBound();
418     } catch (Exception e) {
419       // ignore
420     }
421   }

423   private void closeSocketFactory() {
424     if (socketFactory != null) {
425       try {
426         socketFactory.close();
427       } catch (IOException e) {
428         // ignore
429       }
430     }
431   }

433   private void deleteSocketDir() {
434     if (deleteRegistrySocketDir && registrySocketDir != null) {
435       try {
436         Files.delete(registrySocketDir.toPath());
437       } catch (IOException e) {
438         // ignore
439       }
440     }
441   }

443   private void shutdownViaRMIService(AFUNIXRegistry reg, AFUNIXRMIService serv)
444       throws RemoteException {
445     try {
446       if (serv == null) {
447         serv = getRMIService(reg);
448       }
449       if (serv.isShutdownAllowed()) {
450         serv.shutdown();
451       }
452     } catch (ServerException | ConnectIOException | NotBoundException e) {
453       // ignore
454     }
455   }

457   /**
458    * Creates a new RMI {@link Registry}.
459    * 
460    * If there already was a registry created previously, it is shut down and replaced by the current
461    * one.
462    * 
463    * Use {@link #getRegistry()} to try to reuse an existing registry.
464    * 
465    * @return The registry
466    * @throws RemoteException if the operation fails.
467    * @see #getRegistry()
468    */
469   public AFUNIXRegistry createRegistry() throws RemoteException {
470     synchronized (AFUNIXNaming.class) {
471       AFUNIXRegistry existingRegistry = registry;
472       if (existingRegistry == null) {
473         try {
474           existingRegistry = getRegistry(false);
475         } catch (ServerException e) {
476           Throwable cause = e.getCause();
477           if (cause instanceof NotBoundException || cause instanceof ConnectIOException) {
478             existingRegistry = null;
479           } else {
480             throw e;
481           }
482         }
483       }
484       if (existingRegistry != null) {
485         if (!isRemoteShutdownAllowed()) {
486           throw new ServerException("The server refuses to be shutdown remotely");
487         }
488         shutdownRegistry();
489       }
490 
491       if (registrySocketDir != null && !registrySocketDir.mkdirs() && !registrySocketDir
492           .isDirectory()) {
493         throw new ServerException("Cannot create socket directory:" + registrySocketDir);
494       }
495       setRegistry(new AFUNIXRegistry(this, LocateRegistry.createRegistry(registryPort,
496           socketFactory, socketFactory)));
497 
498       final AFUNIXRMIService service = new AFUNIXRMIServiceImpl(this);
499       UnicastRemoteObject.exportObject(service, servicePort, socketFactory, socketFactory);
500 
501       rebindRMIService(service);
502 
503       return registry;
504     }
505   }

592   private void setRegistry(AFUNIXRegistry registry) {
593     synchronized (AFUNIXNaming.class) {
594       this.registry = registry;
595       if (registry == null) {
596         rmiService = null;
597       } else if (registry.isLocal()) {
598         closeUponRuntimeShutdown();
599       }
600     }
